74int xl320_sendCommand(XL320_t* xl320, XL320_Instruction_t inst, uint16_t nbParams, uint8_t* params){
75 uint8_t* txBuff = NULL;
76 txBuff = (uint8_t*) malloc((MIN_FRAME_SIZE + nbParams)*sizeof(uint8_t));
77 uint16_t length = nbParams + 3;
78
79 xl320_addHeader2Buff(txBuff);
80 txBuff[4] = xl320->id;
81 txBuff[5] = (uint8_t) (length & 0xFF);
82 txBuff[6] = (uint8_t) (length >> 8);
83 txBuff[7] = inst;
84 xl320_copyParams2Buff(8, txBuff, nbParams, params);
85
86 uint16_t crc = xl320_updateCrc(0, txBuff, MIN_FRAME_SIZE + nbParams - CRC_FIELD_SIZE);
87
88 txBuff[(MIN_FRAME_SIZE + nbParams) - 2] = (uint8_t) (crc & 0xFF);
89 txBuff[(MIN_FRAME_SIZE + nbParams) - 1] = (uint8_t) (crc >> 8);
90
91 xl320->serial.transmit(txBuff, (MIN_FRAME_SIZE + nbParams)*sizeof(uint8_t), 0x1F4);
92 /*
93 HAL_HalfDuplex_EnableTransmitter(&huart6);
94 HAL_UART_Transmit(xl320->uart, txBuff, (MIN_FRAME_SIZE + nbParams)*sizeof(uint8_t), 0x1F4);
95 */
96
97 free(txBuff);
98 return 0;
99}

107int xl320_checkErrorField(uint8_t errCode){
108 switch(errCode){
109 case NO_ERROR :
110 return 0;
111 case RESULT_FAIL :
112 DEBUG_PRINTF("XL320 ERROR : Failed to process the sent Instruction Packet\r\n");
113 return -1;
114 case INSTR_ERROR :
115 DEBUG_PRINTF("XL320 ERROR : Undefined Instruction has been used or action has been used without Reg/Write\r\n");
116 return -1;
117 case CRC_ERROR :
118 DEBUG_PRINTF("XL320 ERROR : CRC of the sent Packet does not match\r\n");
119 return -1;
120 case DATA_RANGE_ERROR :
121 DEBUG_PRINTF("XL320 ERROR : Data to be written in the corresponding Address is outside the range of the min/max value\r\n");
122 return -1;
123 case DATA_LENGTH_ERROR :
124 DEBUG_PRINTF("XL320 ERROR : Attempt to write Data that is shorter than the data length of the corresponding Address\r\n");
125 return -1;
126 case DATA_LIMIT_ERROR :
127 DEBUG_PRINTF("XL320 ERROR : Data to be written in the corresponding Address is outside of the Limit value\r\n");
128 return -1;
129 case ACCESS_ERROR :
130 DEBUG_PRINTF("XL320 ERROR : Attempt to write a value in an Address that is Read Only or has not been defined\r\n");
131 DEBUG_PRINTF(" Attempt to read a value in an Address that is Write Only or has not been defined\r\n");
132 DEBUG_PRINTF(" Attempt to write a value in the ROM domain while in a state of Torque Enable(ROM Lock)\r\n");
133 return -1;
134 }
135 return 0;

138int xl320_check_crcField(uint8_t* buffer){
139 uint16_t nbParam = buffer[LEN_FRAME_OFFSET] + (buffer[LEN_FRAME_OFFSET + 1]<<8) - 3;
140
141 uint16_t crcReceived = buffer[MIN_FRAME_SIZE + nbParam - 1] + (buffer[MIN_FRAME_SIZE + nbParam]<<8);
142 uint16_t crcComputed = xl320_updateCrc(0, buffer, MIN_FRAME_SIZE + nbParam - 2);

144 if(crcReceived == crcComputed)
145 return 0;
